// services/chunker.js — 슬라이딩 윈도우
export function chunkText(text, size=650, overlap=100) {
  let start = 0;
  while (start < text.length) {
    let end = start + size;
    // 문장 경계에서 자르기 (가독성 향상)
    const cut = slice.lastIndexOf('. ');
    chunks.push(text.slice(start, end).trim());
    start = Math.max(0, end - overlap); // overlap
  }
  return chunks;
}

// routes/exam.js — 채점 + 이력 저장
router.post('/grade', async (req, res) => {
  const { examType, source, answers } = req.body;
  const correct = answers.filter(a => a.isCorrect).length;
  const attempt = await prisma.examAttempt.create({
    data: {
      examType, source, correct,
      total: answers.length,
      percent: Math.round(correct / answers.length * 100),
      items: { createMany({ data: answers }) }
    }
  });
  res.json(attempt);
});

// services/rag.js
function extractJsonCandidate(raw) {
  const cleaned = raw
    .replace(/```json|```/g, '').trim();
  const start = cleaned.indexOf('{');
  const end   = cleaned.lastIndexOf('}');
  return cleaned.slice(start, end + 1);
}

async function callQuestionsModel(prompt, maxRetry=1) {
  for (let i = 0; i <= maxRetry; i++) {
    const res = await client.chat.completions.create({
      model: CHAT_MODEL,
      messages: [{ role: 'user', content: prompt }],
      max_tokens: 2600,
    });
    const raw = res.choices[0]?.message?.content || '';
    try { return parseQuestions(raw); }
    catch (e) { if (i === maxRetry) throw e; }
  }
}
